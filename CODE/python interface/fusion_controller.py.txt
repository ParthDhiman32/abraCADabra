import pyautogui
import time

class FusionController:
    # Fusion 360 Shortcuts:
    # Rotate: Shift + Middle Mouse Drag
    # Pan: Middle Mouse Drag
    # Zoom: Scroll Wheel
    # Fit: F6

    @staticmethod
    def zoom_in():
        pyautogui.scroll(300)

    @staticmethod
    def zoom_out():
        pyautogui.scroll(-300)

    @staticmethod
    def rotate_left():
        pyautogui.keyDown('shift')
        pyautogui.mouseDown(button='middle')
        pyautogui.move(-100, 0, 0.1) # Smooth sweep
        pyautogui.mouseUp(button='middle')
        pyautogui.keyUp('shift')

    @staticmethod
    def rotate_right():
        pyautogui.keyDown('shift')
        pyautogui.mouseDown(button='middle')
        pyautogui.move(100, 0, 0.1)
        pyautogui.mouseUp(button='middle')
        pyautogui.keyUp('shift')

    @staticmethod
    def reset_view():
        pyautogui.press('f6') # Default Fusion 'Fit' key
