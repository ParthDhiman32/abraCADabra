import serial
import serial.tools.list_ports
import time
import threading
import asyncio
from fusion_controller import FusionController
from media_controller import MediaController

class AppController:
    def __init__(self):
        self.serial_port = None
        self.current_mode = "CAD"
        self.is_running = True

    def auto_connect_serial(self):
        while self.is_running:
            if self.serial_port is None or not self.serial_port.is_open:
                ports = serial.tools.list_ports.comports()
                for port in ports:
                    # Look for standard CH340 / CP210x USB UART typical of ESP32
                    if "USB" in port.description or "CH340" in port.description or "UART" in port.description:
                        try:
                            self.serial_port = serial.Serial(port.device, 115200, timeout=1)
                            print(f"[+] Connected to ESP32 on {port.device}")
                            time.sleep(2) # Wait for ESP32 to reboot after serial connect
                            break
                        except (OSError, serial.SerialException):
                            pass
                if not self.serial_port:
                    print("[-] Waiting for ESP32 connection...")
                    time.sleep(3)
            else:
                time.sleep(1)

    def process_gesture(self, gesture):
        print(f"Action triggered: {self.current_mode} -> {gesture}")
        if self.current_mode == "CAD":
            if gesture == "UP": FusionController.zoom_out()
            elif gesture == "DOWN": FusionController.zoom_in()
            elif gesture == "LEFT": FusionController.rotate_left()
            elif gesture == "RIGHT": FusionController.rotate_right()
            elif gesture == "FORWARD": FusionController.reset_view()
        elif self.current_mode == "MUSIC":
            if gesture == "UP": MediaController.vol_up()
            elif gesture == "DOWN": MediaController.vol_down()
            elif gesture == "LEFT": MediaController.prev_track()
            elif gesture == "RIGHT": MediaController.next_track()
            elif gesture == "WAVE": MediaController.play_pause()

    def serial_listener(self):
        while self.is_running:
            if self.serial_port and self.serial_port.is_open:
                try:
                    if self.serial_port.in_waiting:
                        line = self.serial_port.readline().decode('utf-8').strip()
                        if line.startswith("MODE:"):
                            self.current_mode = line.split(":")[1]
                            print(f"[*] Mode switched to {self.current_mode}")
                        elif line.startswith("GESTURE:"):
                            gesture = line.split(":")[1]
                            self.process_gesture(gesture)
                except Exception as e:
                    print(f"[-] Serial error: {e}. Disconnecting.")
                    self.serial_port.close()
                    self.serial_port = None
            else:
                time.sleep(0.5)

    async def media_broadcaster(self):
        """Sends metadata to ESP32 every second."""
        while self.is_running:
            if self.serial_port and self.serial_port.is_open and self.current_mode == "MUSIC":
                media_info = await MediaController.get_current_media_info()
                try:
                    self.serial_port.write(f"SONG:{media_info}\n".encode('utf-8'))
                    # Fake a volume level for demo UI purposes (winsdk volume control requires CoreAudio)
                    self.serial_port.write(f"VOL:75\n".encode('utf-8')) 
                except Exception:
                    pass
            await asyncio.sleep(1)

    def run(self):
        print("Starting Smart Gesture Desktop Service...")
        threading.Thread(target=self.auto_connect_serial, daemon=True).start()
        threading.Thread(target=self.serial_listener, daemon=True).start()
        
        # Run asyncio loop for media fetching
        loop = asyncio.get_event_loop()
        loop.run_until_complete(self.media_broadcaster())

if __name__ == "__main__":
    app = AppController()
    try:
        app.run()
    except KeyboardInterrupt:
        print("\nExiting...")
        app.is_running = False
